CREATE OR REPLACE PROCEDURE szamok(n number) IS 
  
  faktor integer := 1;
  
BEGIN
  
  BEGIN
    dbms_output.put_line(1/n);
  EXCEPTION
    when others then
      dbms_output.put_line(sqlerrm);
  END;
  
  BEGIN
    dbms_output.put_line(sqrt(n));
  EXCEPTION
    when others then
      dbms_output.put_line(sqlerrm);
  END;

  BEGIN
    IF n >= 0 AND n <= 1 THEN
      dbms_output.put_line(1);
    ELSE
      IF n > 0 THEN
        FOR i IN 1 .. n LOOP
          faktor := faktor * i;
        END LOOP;
        dbms_output.put_line(faktor);
      ELSE
        RAISE value_error;
      END IF;  
    END IF;
    
    EXCEPTION
      when others then
        dbms_output.put_line(sqlerrm); 
  END;
  
END;
